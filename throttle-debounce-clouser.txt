function debounce(fn, delay) {
  let timer;

  return function () {
    clearTimeout(timer);

    timer = setTimeout(() => {
      fn();
    }, delay);
  };
}

  console.log("API Call");

function throttle(fn, limit) {
  let flag = true;

  return function () {
    if (flag) {
      fn();
      flag = false;

      setTimeout(() => {
        flag = true;
      }, limit);
    }
  };
}

function onScroll() {
  console.log("Scroll event fired");
}

let throttleScroll = throttle(onScroll, 1000);
